## How to Use This Data

### Matching to Your Database

All problem slugs listed above correspond to the `title_slug` field in your `problems` table. To query problems for a specific list:

```sql
-- Get all Blind 75 problems
SELECT * FROM problems
WHERE title_slug IN (
  'two-sum',
  'contains-duplicate',
  -- ... rest of slugs
);

-- Get problems by category for Blind 75
SELECT p.*
FROM study_plan_template_milestones m
JOIN LATERAL unnest(m.problem_ids) AS pid ON true
JOIN problems p ON p.id = pid
WHERE m.template_id = (
  SELECT id FROM study_plan_templates
  WHERE display_name = 'Blind 75 Essentials'
)
ORDER BY m.milestone_order, p.id;
```

### Creating a Study Plan Script

To create a TypeScript/JavaScript script that fetches and organizes these problems:

```typescript
import { createClient } from '@supabase/supabase-js';

const supabase = createClient(url, key);

// Get all problems for a study plan template
async function getStudyPlanProblems(templateName: string) {
  const { data: milestones } = await supabase
    .from('study_plan_template_milestones')
    .select(`
      id,
      title,
      description,
      milestone_order,
      problem_ids,
      template:study_plan_templates!inner(display_name)
    `)
    .eq('template.display_name', templateName)
    .order('milestone_order');

  // Fetch all problems for each milestone
  for (const milestone of milestones) {
    const { data: problems } = await supabase
      .from('problems')
      .select('*')
      .in('id', milestone.problem_ids);

    milestone.problems = problems;
  }

  return milestones;
}

// Usage
const blind75 = await getStudyPlanProblems('Blind 75 Essentials');
console.log(blind75);
```
